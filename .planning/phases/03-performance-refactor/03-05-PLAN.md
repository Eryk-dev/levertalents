**1. `supabase/tests/003_evaluation_cycles_snapshot.sql`** (substitui skip(8) por 8 tests):
```sql
BEGIN;
CREATE EXTENSION IF NOT EXISTS pgtap;
SELECT plan(8);

-- INV-3-01 + perf1
SELECT has_table('public', 'evaluation_cycles', 'evaluation_cycles table exists');
SELECT has_table('public', 'evaluation_templates', 'evaluation_templates table exists');
SELECT col_not_null('public', 'evaluation_cycles', 'company_id', 'cycles.company_id NOT NULL');
SELECT col_not_null('public', 'evaluation_cycles', 'template_snapshot', 'cycles.template_snapshot NOT NULL');

-- INV-3-05: snapshot freeze trigger
DO $$
DECLARE
  v_cid uuid := '00000000-0000-0000-0000-000000000001';
  v_tid uuid := '00000000-0000-0000-0000-000000000002';
  v_company uuid;
  v_orig jsonb;
  v_after jsonb;
BEGIN
  SELECT id INTO v_company FROM public.companies LIMIT 1;
  IF v_company IS NULL THEN RAISE NOTICE 'No companies — skip'; RETURN; END IF;

  INSERT INTO public.evaluation_templates (id, company_id, name, schema_json) VALUES (v_tid, v_company, 'T1', '{"version":1,"sections":[]}'::jsonb);
  INSERT INTO public.evaluation_cycles (id, company_id, template_id, name, template_snapshot, starts_at, ends_at)
    VALUES (v_cid, v_company, v_tid, 'C1', '{"hijack":true}'::jsonb, NOW(), NOW() + interval '1 day');

  -- Trigger should have OVERWRITTEN template_snapshot to schema_json
  SELECT template_snapshot INTO v_orig FROM public.evaluation_cycles WHERE id = v_cid;

  -- Now mutate template
  UPDATE public.evaluation_templates SET schema_json = '{"version":1,"sections":[{"id":"new","title":"new","weight":1,"questions":[]}]}'::jsonb WHERE id = v_tid;

  -- Cycle snapshot should be UNCHANGED
  SELECT template_snapshot INTO v_after FROM public.evaluation_cycles WHERE id = v_cid;
  PERFORM ok(v_orig = v_after, 'INV-3-05: template_snapshot frozen — UPDATE on template does NOT affect existing cycle');
END $$;

-- INV-3-05 (negative): direct UPDATE of snapshot raises exception
SELECT throws_ok(
  $$ UPDATE public.evaluation_cycles SET template_snapshot = '{"tampered":true}'::jsonb WHERE id = '00000000-0000-0000-0000-000000000001' $$,
  'P0001',
  'template_snapshot is immutable after cycle creation',
  'UPDATE template_snapshot raises exception'
);

-- INV-3-23: cron job registered
SELECT ok(
  EXISTS (SELECT 1 FROM cron.job WHERE jobname = 'evaluation_cycles_auto_close'),
  'INV-3-23: cron job evaluation_cycles_auto_close registered'
);

-- Cleanup
DELETE FROM public.evaluation_cycles WHERE id = '00000000-0000-0000-0000-000000000001';
DELETE FROM public.evaluation_templates WHERE id = '00000000-0000-0000-0000-000000000002';

-- Schedule check
SELECT is(
  (SELECT schedule FROM cron.job WHERE jobname = 'evaluation_cycles_auto_close'),
  '0 6 * * *',
  'cron job schedule is 0 6 * * * (06:00 UTC = 03:00 BRT)'
);

SELECT * FROM finish();
ROLLBACK;
```
