import { useMutation, useQueryClient } from '@tanstack/react-query';
import { supabase } from '@/integrations/supabase/client';
import { useScopedQuery } from '@/shared/data/useScopedQuery';
import { scopeKey } from '@/lib/scopeKey';
import { useScope } from '@/shared/data/useScope';
import type { Database } from '@/integrations/supabase/types';

type EvaluationRow = Database['public']['Tables']['evaluations']['Row'];
type EvaluationInsert = Database['public']['Tables']['evaluations']['Insert'];

export type EvaluationDirection = 'leader_to_member' | 'member_to_leader';

export interface CreateEvaluationInput {
  cycle_id: string;
  evaluator_user_id: string;
  evaluated_user_id: string;
  direction: EvaluationDirection;
  responses: Record<string, unknown>; // shape derived from template_snapshot
}

export function useEvaluations(cycleId: string | null) {
  return useScopedQuery(
    ['evaluations', cycleId],
    async (scope) => {
      if (!cycleId) return [] as EvaluationRow[];
      let query = supabase
        .from('evaluations')
        .select('*')
        .eq('cycle_id', cycleId)
        .order('updated_at', { ascending: false });
      // RLS handles visibility (D-03); company_id filter is defense-in-depth
      const { data, error } = await query;
      if (error) throw error;
      return (data ?? []) as EvaluationRow[];
    },
    { enabled: cycleId != null },
  );
}

export function useCreateEvaluation() {
  const queryClient = useQueryClient();
  const scope = useScope();
  return useMutation({
    mutationFn: async (input: CreateEvaluationInput) => {
      // Pitfall §1: ensure company_id is set (NOT NULL post-Plan 03-04 perf2).
      // Resolve via cycle.company_id (cycle is scoped — RLS guarantees current scope).
      const { data: cycle, error: cyErr } = await supabase
        .from('evaluation_cycles')
        .select('company_id')
        .eq('id', input.cycle_id)
        .single();
      if (cyErr || !cycle) throw cyErr ?? new Error('Cycle not found');

      const insert: EvaluationInsert = {
        cycle_id: input.cycle_id,
        evaluator_user_id: input.evaluator_user_id,
        evaluated_user_id: input.evaluated_user_id,
        direction: input.direction,
        responses: input.responses,
        company_id: cycle.company_id,
      };
      const { data, error } = await supabase
        .from('evaluations')
        .insert(insert)
        .select()
        .single();
      if (error) throw error;
      return data as EvaluationRow;
    },
    onSuccess: (_data, vars) => {
      queryClient.invalidateQueries({ queryKey: scopeKey(scope, 'evaluations', vars.cycle_id) });
    },
  });
}

export function useUpdateEvaluation() {
  const queryClient = useQueryClient();
  const scope = useScope();
  return useMutation({
    mutationFn: async (vars: { id: string; cycle_id: string; responses: Record<string, unknown> }) => {
      const { data, error } = await supabase
        .from('evaluations')
        .update({ responses: vars.responses, updated_at: new Date().toISOString() })
        .eq('id', vars.id)
        .select()
        .single();
      if (error) throw error;
      return data as EvaluationRow;
    },
    onSuccess: (_data, vars) => {
      queryClient.invalidateQueries({ queryKey: scopeKey(scope, 'evaluations', vars.cycle_id) });
    },
  });
}

  name: string;
  starts_at: string; // ISO
  ends_at: string;   // ISO; must be > starts_at (CHECK in DB)

export function useEvaluationCycles() {
  return useScopedQuery(
    ['evaluation_cycles'],
    async (scope) => {
      const ids = scope.companyIds;
      if (!ids.length) return [] as CycleRow[];
      const { data, error } = await supabase

        .select('*')

  const scope = useScope();
  return useMutation({
    mutationFn: async (input: CreateCycleInput) => {
      if (new Date(input.ends_at) <= new Date(input.starts_at)) {
        throw new Error('A data de fim precisa ser depois da data de início.');

        company_id: input.company_id,
        template_id: input.template_id,
        name: input.name,
        starts_at: input.starts_at,
        ends_at: input.ends_at,
        template_snapshot: {} as never, // overwritten by trigger
        status: 'active',
      };
      const { data, error } = await supabase
        .from('evaluation_cycles')
        .insert(insert)
        .select()
        .single();
      if (error) throw error;
      return data as CycleRow;
    },
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: scopeKey(scope, 'evaluation_cycles') });
    },

import { useMutation, useQueryClient } from '@tanstack/react-query';
import { supabase } from '@/integrations/supabase/client';
import { useScopedQuery } from '@/shared/data/useScopedQuery';
import { scopeKey } from '@/lib/scopeKey';
import { useScope } from '@/shared/data/useScope';
import type { Database } from '@/integrations/supabase/types';

type OneOnOneRow = Database['public']['Tables']['one_on_ones']['Row'];

export interface OneOnOneFilters {
  leaderId?: string;
  collaboratorId?: string;
  status?: 'scheduled' | 'in-progress' | 'completed' | 'processing';
  fromDate?: string; // ISO
  toDate?: string;
}

export function useOneOnOnes(filters: OneOnOneFilters = {}) {
  return useScopedQuery(
    ['one_on_ones', filters],
    async (scope) => {
      const ids = scope.companyIds;
      if (!ids.length) return [] as OneOnOneRow[];
      let q = supabase
        .from('one_on_ones')
        .select('*')
        .in('company_id', ids)
        .order('scheduled_at', { ascending: false });
      if (filters.leaderId)        q = q.eq('leader_id', filters.leaderId);
      if (filters.collaboratorId)  q = q.eq('collaborator_id', filters.collaboratorId);
      if (filters.status)          q = q.eq('status', filters.status);
      if (filters.fromDate)        q = q.gte('scheduled_at', filters.fromDate);
      if (filters.toDate)          q = q.lte('scheduled_at', filters.toDate);
      const { data, error } = await q;
      if (error) throw error;
      return (data ?? []) as OneOnOneRow[];
    },
  );
}

export interface UpdateOneOnOneInput {
  id: string;
  meeting_structure?: Record<string, unknown>; // includes transcricao_plaud, resumo_plaud, agenda_items, action_items
  status?: 'scheduled' | 'in-progress' | 'completed' | 'processing';
}

export function useUpdateOneOnOne() {
  const queryClient = useQueryClient();
  const scope = useScope();
  return useMutation({
    mutationFn: async (input: UpdateOneOnOneInput) => {
      const update: Record<string, unknown> = { updated_at: new Date().toISOString() };
      if (input.meeting_structure !== undefined) update.meeting_structure = input.meeting_structure;
      if (input.status !== undefined) update.status = input.status;
      const { data, error } = await supabase
        .from('one_on_ones')
        .update(update)
        .eq('id', input.id)
        .select()
        .single();
      if (error) throw error;
      return data as OneOnOneRow;
    },
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: scopeKey(scope, 'one_on_ones') });
    },
  });
}

import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query';

        .from('one_on_one_rh_notes')

  const queryClient = useQueryClient();
  const scope = useScope();
  return useMutation({
    mutationFn: async (input: { meeting_id: string; notes: string | null }) => {

import { useMutation, useQueryClient } from '@tanstack/react-query';
import { supabase } from '@/integrations/supabase/client';

export function useChangePassword() {
  const queryClient = useQueryClient();
  return useMutation({
    mutationFn: async (input: { newPassword: string }) => {
      // Step 1: update auth password
      const { data: { user } } = await supabase.auth.getUser();
      if (!user) throw new Error('Não autenticado');

      const { error: pwdError } = await supabase.auth.updateUser({ password: input.newPassword });
      if (pwdError) throw pwdError;

      // Step 2: flip must_change_password flag in profiles (Pitfall §9: not atomic; retry on failure)
      const { error: profileError } = await supabase
        .from('profiles')
        .update({
          must_change_password: false,
          temp_password_expires_at: null,
        })
        .eq('id', user.id);

      if (profileError) {
        // Retry once
        const { error: retryError } = await supabase
          .from('profiles')
          .update({ must_change_password: false, temp_password_expires_at: null })
          .eq('id', user.id);
        if (retryError) throw retryError;
      }

      return { success: true };
    },
    onSuccess: async () => {
      // Pitfall §9: MUST invalidate profile cache so ProtectedRoute re-reads must_change_password=false
      await queryClient.invalidateQueries({ queryKey: ['userProfile'] });
      // Wait for refetch to complete before redirect (handled in component)
    },
  });
}
