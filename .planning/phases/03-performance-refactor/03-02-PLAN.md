```sql
-- =========================================================================
-- Migration E2: teams → org_units backfill (1:1 com members + leaders)
--
-- Threats: T-3-06 (cross-tenant leak via missing company_id mapping — covered by NOT EXISTS guard)
-- REQs: ORG-09 (teams legacy permanece read-only) + Phase 3 D-28
-- Reversibility: idempotent — INSERT ... WHERE NOT EXISTS; re-run produces 0 changes
-- DEPENDENCIES: Migration E1 must be applied first (companies + group exist)
-- NOTE: teams table is preserved (ORG-09 read-only); drops only in Phase 4 Migration G
-- =========================================================================

-- Step 1 — Ensure each company has a root org_unit (parent_id IS NULL).
-- Phase 1 Migration B may have created these; this is defensive.
INSERT INTO public.org_units (id, company_id, parent_id, name, kind, position, created_at)
SELECT gen_random_uuid(), c.id, NULL, c.name || ' (raiz)', 'organizacao', 0, NOW()
  FROM public.companies c
 WHERE NOT EXISTS (SELECT 1 FROM public.org_units ou WHERE ou.company_id = c.id AND ou.parent_id IS NULL);

-- Step 2 — Convert each team into 1 org_unit (preserving team.id as org_unit.id)
INSERT INTO public.org_units (id, company_id, parent_id, name, kind, position, created_at)
SELECT t.id,
       t.company_id,
       (SELECT ou.id FROM public.org_units ou
          WHERE ou.company_id = t.company_id AND ou.parent_id IS NULL
          LIMIT 1),
       t.name,
       'time',
       0,
       COALESCE(t.created_at, NOW())
  FROM public.teams t
 WHERE NOT EXISTS (SELECT 1 FROM public.org_units ou WHERE ou.id = t.id);

-- Step 3 — Convert team_members → org_unit_members (1:1 user_id + team_id)
INSERT INTO public.org_unit_members (user_id, org_unit_id, joined_at)
SELECT tm.user_id, tm.team_id, COALESCE(tm.created_at, NOW())
  FROM public.team_members tm
 WHERE EXISTS (SELECT 1 FROM public.org_units ou WHERE ou.id = tm.team_id)
   AND NOT EXISTS (
     SELECT 1 FROM public.org_unit_members oum
      WHERE oum.user_id = tm.user_id AND oum.org_unit_id = tm.team_id
   );

-- Step 4 — Convert teams.leader_id → unit_leaders (single leader per team in legacy → 1 row)
INSERT INTO public.unit_leaders (user_id, org_unit_id, assigned_at)
SELECT t.leader_id::uuid, t.id, NOW()
  FROM public.teams t
 WHERE t.leader_id IS NOT NULL
   AND t.leader_id ~ '^[0-9a-f]{8}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{12}$'  -- defensive UUID check (teams.leader_id is TEXT)
   AND EXISTS (SELECT 1 FROM auth.users au WHERE au.id = t.leader_id::uuid)
   AND NOT EXISTS (
     SELECT 1 FROM public.unit_leaders ul
      WHERE ul.user_id = t.leader_id::uuid AND ul.org_unit_id = t.id
   );

COMMENT ON TABLE public.org_units IS 'Phase 1 schema; backfilled from teams in Phase 3 Migration E2';
```

- `auth.users` join garante leader ainda existe (leader pode ter sido excluído pós-cadastro de team)
- `NOT EXISTS` em todas inserts garante idempotência
