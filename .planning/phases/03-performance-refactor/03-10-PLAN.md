**1. Crie `src/components/ClimateAggregateCard.tsx`:**
```tsx
import { Card, SectionHeader } from '@/components/primitives/LinearKit';
import { Skeleton } from '@/components/ui/skeleton';
import { Users } from 'lucide-react';
import { useClimateAggregate } from '@/hooks/useClimateAggregate';

export interface ClimateAggregateCardProps {
  surveyId: string;
  orgUnitId?: string | null;
  surveyName: string;
  orgUnitName?: string;
}

export function ClimateAggregateCard({ surveyId, orgUnitId = null, surveyName, orgUnitName }: ClimateAggregateCardProps) {
  const { data, isLoading } = useClimateAggregate(surveyId, orgUnitId);

  if (isLoading) {
    return (
      <Card className="min-h-[160px]">
        <Skeleton className="h-4 w-2/3 mb-2" />
        <Skeleton className="h-8 w-full" />
      </Card>
    );
  }

  // K-anonymity empty state — Pitfall §3: NEVER show count when insufficient
  if (!data || ('insufficient_data' in data && data.insufficient_data)) {
    return (
      <Card className="min-h-[160px] flex flex-col items-center justify-center text-center">
        <Users className="size-6 text-text-subtle mb-2" aria-hidden="true" />
        <SectionHeader>Dados insuficientes para garantir anonimato</SectionHeader>
        <p className="text-sm text-text-subtle mt-1 max-w-xs">
          Aguarde no mínimo 3 respostas para esta unidade. Esta pesquisa é 100% anônima.
        </p>
        {/* No CTA — wait state */}
      </Card>
    );
  }

  return (
    <Card className="min-h-[160px]">
      <div className="flex items-baseline justify-between mb-2">
        <SectionHeader>{surveyName}</SectionHeader>
        {orgUnitName && <span className="text-xs text-text-subtle">{orgUnitName}</span>}
      </div>
      <div className="flex items-baseline gap-3">
        <span className="text-display-md tabular">{data.avg.toFixed(1)}</span>
        <span className="text-xs text-text-subtle">média ({data.count} respostas)</span>
      </div>
      <div className="mt-3 grid grid-cols-5 gap-1">
        {([1, 2, 3, 4, 5] as const).map((score) => {
          const cnt = data.distribution[String(score)] ?? 0;
          const pct = data.count > 0 ? (cnt / data.count) * 100 : 0;
          return (
            <div key={score} className="flex flex-col items-center">
              <div className="w-full bg-bg-subtle rounded relative h-12">
                <div className="absolute bottom-0 left-0 right-0 bg-accent rounded" style={{ height: `${pct}%` }} />
              </div>
              <span className="text-xs mt-1">{score}</span>
            </div>
          );
        })}
      </div>
    </Card>
  );
}
```

import { describe, it, expect, vi } from 'vitest';
import { render, screen } from '@testing-library/react';
import { ClimateAggregateCard } from '../ClimateAggregateCard';

const mockHook = vi.fn();
vi.mock('@/hooks/useClimateAggregate', () => ({

describe('ClimateAggregateCard k-anon (D-10) [INV-3-09]', () => {
  it('renders empty state when insufficient_data:true', () => {
    mockHook.mockReturnValue({ data: { insufficient_data: true }, isLoading: false });

    expect(screen.getByText('Dados insuficientes para garantir anonimato')).toBeInTheDocument();
    expect(screen.getByText(/100% anônima/i)).toBeInTheDocument();

    expect(screen.queryByText(/1 resposta/i)).not.toBeInTheDocument();
    expect(screen.queryByText(/2 respostas/i)).not.toBeInTheDocument();
    // No CTA — wait state
    expect(screen.queryByRole('button')).not.toBeInTheDocument();
  });

  it('renders full aggregate when count >= 3', () => {
    mockHook.mockReturnValue({
      data: { count: 5, avg: 4.2, distribution: { '1': 0, '2': 0, '3': 1, '4': 2, '5': 2 } },

    render(<ClimateAggregateCard surveyId="s1" surveyName="Pesquisa Q1" />);
    expect(screen.getByText('4.2')).toBeInTheDocument();
    expect(screen.getByText(/5 respostas/i)).toBeInTheDocument();

  it('renders skeleton while loading', () => {
    mockHook.mockReturnValue({ data: undefined, isLoading: true });
    const { container } = render(<ClimateAggregateCard surveyId="s1" surveyName="Pesquisa Q1" />);

import { useState } from 'react';
import { Btn, SectionHeader } from '@/components/primitives/LinearKit';
import { Info } from 'lucide-react';
import { ClimateAggregateCard } from '@/components/ClimateAggregateCard';
import { ClimateSurveyFormDialog } from '@/components/ClimateSurveyFormDialog';

export default function ClimatePage() {

  return (

    </div>
  );
}
