import { useEffect } from "react";
import { useSearchParams } from "react-router-dom";
import { useCandidate } from "@/hooks/hiring/useCandidates";
import { useScope } from "@/app/providers/ScopeProvider";
import { useAbility } from "@/lib/abilities"; // Phase 1 CASL
import { CandidateDrawerHeader } from "./CandidateDrawerHeader";
import { CandidateDrawerTabs, type DrawerTab } from "./CandidateDrawerTabs";
import { CandidateDrawerContent } from "./CandidateDrawerContent";
import type { ApplicationWithCandidate } from "@/integrations/supabase/hiring-types";

interface CandidateDrawerProps {
  candidateId: string;
  application?: ApplicationWithCandidate | null;
  onClose: () => void;
}

export function CandidateDrawer({ candidateId, application, onClose }: CandidateDrawerProps) {
  const [searchParams, setSearchParams] = useSearchParams();
  const activeTab = (searchParams.get("tab") as DrawerTab) ?? "perfil";

  const { data: candidate, isLoading } = useCandidate(candidateId, "drawer");
  const ability = useAbility();
  const showAuditLog = ability?.can("read", "DataAccessLog") ?? false;

  const setActiveTab = (tab: DrawerTab) => {
    setSearchParams((prev) => {
      const next = new URLSearchParams(prev);
      next.set("tab", tab);
      return next;
    }, { replace: true });
  };

  // ESC handler — fecha sem perder scroll do board parent
  useEffect(() => {
    const onEsc = (e: KeyboardEvent) => {
      if (e.key === "Escape") {
        e.stopPropagation();
        onClose();
      }
    };
    window.addEventListener("keydown", onEsc);
    return () => window.removeEventListener("keydown", onEsc);
  }, [onClose]);

  if (isLoading) return <aside className="w-[480px] border-l border-default bg-surface" />;
  if (!candidate) return null;

  return (
    <aside
      className="flex h-full flex-col w-full md:w-[480px] shrink-0 border-l border-default bg-surface overflow-hidden"
      role="complementary"
      aria-label="Detalhes do candidato"
    >
      <CandidateDrawerHeader
        candidate={candidate}
        application={application}
        onClose={onClose}
        onMove={() => {/* TODO: integrar com move dialog */}}
        onScheduleInterview={() => {/* TODO */}}
        onRequestFit={() => {/* TODO */}}
      />
      <CandidateDrawerTabs
        activeTab={activeTab}
        onTabChange={setActiveTab}
        showAuditLog={showAuditLog}
      />
      <CandidateDrawerContent
        activeTab={activeTab}
        candidate={candidate}
        application={application}
      />
    </aside>
  );
}

  candidateName: string;
