import { vi } from "vitest";

type ChannelHandler = (payload: { new: unknown; old?: unknown; eventType: string }) => void;

interface MockChannel {
  on: ReturnType<typeof vi.fn>;
  subscribe: ReturnType<typeof vi.fn>;
  unsubscribe: ReturnType<typeof vi.fn>;
  // helpers para o test driver
  __emit: (handlerKey: string, payload: { new: unknown; old?: unknown; eventType: string }) => void;
  __handlers: Record<string, ChannelHandler>;
}

/**
 * Cria um fake de supabase.channel(name) que captura .on(...) handlers
 * e expõe __emit() para o teste disparar postgres_changes events.
 *
 * Uso:
 *   const channel = createMockChannel();
 *   vi.mocked(supabase.channel).mockReturnValue(channel as any);
 *   // ... mount hook
 *   channel.__emit("UPDATE", { new: { id: "app-1", stage: "decisao", ... }, eventType: "UPDATE" });
 */
export function createMockChannel(): MockChannel {
  const handlers: Record<string, ChannelHandler> = {};
  const channel: MockChannel = {
    __handlers: handlers,
    on: vi.fn((_event: string, config: { event: string; table: string; filter: string }, handler: ChannelHandler) => {
      handlers[config.event] = handler;
      return channel;
    }),
    subscribe: vi.fn(() => channel),
    unsubscribe: vi.fn(() => Promise.resolve("ok" as const)),
    __emit: (key, payload) => handlers[key]?.(payload),
  };
  return channel;
}

/**
 * Stub para supabase.removeChannel que registra a chamada para asserts de cleanup.
 */
export function createRemoveChannelSpy() {
  return vi.fn().mockResolvedValue("ok" as const);
}
